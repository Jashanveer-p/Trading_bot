import numpy as np
import pandas as pd
import yfinance as yf
import matplotlib.pyplot as plt
from ta.momentum import RSIIndicator

class TradingBot:
    def __init__(self, symbol, short=20, long=50, rsi_window=14, weights=None):
        self.symbol     = symbol
        self.short      = short
        self.long       = long
        self.rsi_window = rsi_window
        self.weights    = weights or {'ema':0.5, 'rsi':0.3, 'vol':0.2}
        self.data       = pd.DataFrame()

    def fetch_data(self, start, end):
        df = yf.download(
            self.symbol, start=start, end=end, progress=False,
            auto_adjust=False    # preserve raw OHLC
        )
        # flatten MultiIndex if present
        if isinstance(df.columns, pd.MultiIndex):
            df.columns = df.columns.droplevel(1)
        self.data = df

    def compute_indicators(self):
        df = self.data
        # EMAs
        df['EMA_short'] = df['Close'].ewm(span=self.short, adjust=False).mean()
        df['EMA_long']  = df['Close'].ewm(span=self.long,  adjust=False).mean()
        # RSI
        df['RSI']       = RSIIndicator(close=df['Close'], window=self.rsi_window).rsi()
        # 20-day annualised volatility
        df['volatility']= df['Close'].pct_change().rolling(20).std() * np.sqrt(252)
        self.data = df

    def generate_signals(self):
        df = self.data
        # 1/0 long signal from EMA crossover
        df['signal_ema'] = np.where(df['EMA_short'] > df['EMA_long'], 1, 0)
        # confidence components
        # normalize RSI between 30–70
        df['norm_rsi'] = ((df['RSI'] - 30) / 40).clip(0,1)
        # normalize vol (high vol → low confidence)
        df['norm_vol']= (1 - df['volatility']/df['volatility'].max()).clip(0,1)
        # composite score
        w = self.weights
        df['confidence'] = (
              df['signal_ema'] * w['ema']
            + df['norm_rsi']   * w['rsi']
            + df['norm_vol']   * w['vol']
        ).clip(0,1)
        self.data = df

    def backtest(self, capital, risk_per_trade):
        df = self.data.copy()
        df['returns']       = df['Close'].pct_change()
        # use previous day’s signal
        df['strategy_ret']  = df['signal_ema'].shift(1) * df['returns']
        df['equity_curve']  = (1 + df['strategy_ret']).cumprod() * capital
        # store backtest results
        self.data = df
        self.capital = capital

    def metrics(self):
        df = self.data.dropna(subset=['strategy_ret','equity_curve'])
        rets = df['strategy_ret']
        # total return
        total = df['equity_curve'].iloc[-1]/self.capital - 1
        # CAGR
        days = (df.index[-1] - df.index[0]).days
        cagr = (1+total)**(365/days) - 1
        # ann vol & Sharpe
        ann_vol = rets.std() * np.sqrt(252)
        sharpe  = rets.mean()/rets.std() * np.sqrt(252)
        # max drawdown
        roll_max = df['equity_curve'].cummax()
        mdd = ((df['equity_curve'] - roll_max)/roll_max).min()

        print(f"Total Return: {total:.2%}")
